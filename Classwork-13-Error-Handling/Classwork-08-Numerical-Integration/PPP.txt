# Numerical Integration Tool
# INPUT

GET left_limit FROM user
GET right_limit FROM user
GET function_string FROM user
GET integration_method FROM user

intervals <- 1000
step_size <- (right_limit - left_limit) / intervals
total_area <- 0.0

# PROCESS

IF integration_method == "LRM" THEN
    offset <- 0.0
ELSE IF integration_method == "RRM" THEN
    offset <- step_size
ELSE IF integration_method == "MRM" THEN
    offset <- step_size / 2.0
END IF

FOR step_index FROM 0 TO (intervals - 1) DO
    current_x <- left_limit + (step_index * step_size) + offset
    
    # Evaluate function at current x point
    current_height <- EVALUATE function_string AT current_x
    
    total_area <- total_area + (current_height * step_size)
END FOR

# OUTPUT

DISPLAY "The numerical integration result is: " + total_area<

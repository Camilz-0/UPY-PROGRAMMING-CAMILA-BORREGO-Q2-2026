# INPUT
Open "config.txt" for reading
Initialize empty dictionary config
For each line in file:
    Strip whitespace and split line by "=" into key and value
    Convert value to number and store in config with key
Close file

Read width from config["ancho"]
Read height from config["alto"]
Read max_iterations from config["max_iter"]

Open "mandelbrot.csv" for writing
Write header row "fila,columna,iteraciones" to CSV file

# PROCESS & OUTPUT
For each row from 0 to height - 1:
    For each column from 0 to width - 1:
        # Map pixel to complex plane
        real_part ← config["real_min"] + (column / width) * (config["real_max"] - config["real_min"])
        imag_part ← config["imag_min"] + (row / height) * (config["imag_max"] - config["imag_min"])
        c ← complex number from real_part and imag_part
        
        # Mandelbrot iteration
        z ← complex number (0, 0)
        iterations ← 0
        
        While absolute value of z <= 2 and iterations < max_iterations:
            z ← (z * z) + c
            iterations ← iterations + 1
            
        # Output inside the loops
        Write row, column, and iterations to CSV file

Close CSV file
